def readHand(region):

    hand = []
    for suit in ["s", "h", "d", "c"]:
        for rank in ["01", "02", "03", "04", "05", "06", "07", "08", "09", "10", "11", "12", "13"]:
            m = region.exists(suit + rank + , 0.001)
            if m:
                if m.getScore() > 0.99:
                    hand.append([suit, rank, m.getTarget()])
                    if len(hand) == 5:
                        return Hand(hand)
    return Hand(hand)
